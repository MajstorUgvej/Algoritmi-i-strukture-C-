#include <iostream>
#include <vector>

int* Dijakstra_Alg(const std::vector<std::vector<int>>& M, int s, int n)
{
	int* closest = new int[n + 1];
	std::vector<bool> visited(n + 1, false);
	int brV = 1;

	for (int i = 0; i <= n;i++)
		closest[i] = M[s][i];

	closest[s] = 0;
	visited[s] = true;

	while (brV < n - 1)
	{
		int min = INT_MAX, ind = 0;
		for (int i = 1; i <= n; i++)
		{
			if (closest[i] < min && !visited[i])
			{
				min = closest[i];
				ind = i;
			}
		}
		visited[ind] = true; brV++;

		for (int i = 1; i <= n; i++)
		{
			if (!visited[i] && M[ind][i] != INT_MAX && M[ind][i] + min < closest[i])
				closest[i] = M[ind][i] + min;
		}
	}
	return closest;
}

int main() {
	int n = 6;
	std::vector<std::vector<int>> M((n + 1), std::vector<int> ((n + 1), INT_MAX));
	M[1][2] = M[2][1] = 5;
	M[1][3] = M[3][1] = 2;
	M[2][3] = M[3][2] = 1;
	M[2][4] = M[4][2] = 12;
	M[2][5] = M[5][2] = 6;
	M[3][6] = M[6][3] = 4;
	M[4][5] = M[5][4] = 7;
	M[5][6] = M[6][5] = 3;

	int* res = Dijakstra_Alg(M, 1, n);

	for (int i = 1; i <= n; i++)
	{
		std::cout << res[i] << " ";
	}

	delete[]res;
}
