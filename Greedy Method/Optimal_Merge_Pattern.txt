#include <iostream>

void Bubble_Sort_k(int A[], int n, int k)
{
	for (int j = 0; j < k; j++)
	{
		for (int i = 0; i < n - 1; i++)
		{
			if (A[i] < A[i + 1])
				std::swap(A[i], A[i + 1]);
		}
	}
}

int Optimal_Merge_Pattern(int A[], int n)
{
	int time = 0;
	int newMerged = 0;

	while (n > 1)
	{
		
		Bubble_Sort_k(A, n, 2);
		newMerged = A[n - 1] + A[n - 2];
		time += newMerged;
		A[n - 2] = newMerged;
		n--;
	}

	return time;

}



int main()
{
	int A[] = { 2,5,4,1,7,6,8 };
	int n = std::size(A);

	std::cout << "Time: " << Optimal_Merge_Pattern(A, n);
}
