#include <iostream>
#include <string>

using namespace std;

void Permutacija(string rec, int A[], string res = "")
{
	if (res.length() == rec.length())
	{
		cout << res << endl;
	}

	for (int i = 0, n = rec.length(); i < n; i++)
	{
		if (A[i] == 0)
		{
			res += rec[i];
			A[i] = 1;
			Permutacija(rec, A, res);
			A[i] = 0;
			res.pop_back();
		}
	}
}

int main()
{
	string test;
	cin >> test;
	int n = test.length();
	int* A = new int[n];
	for (int i = 0; i < n; i++)
		A[i] = 0;
	Permutacija(test,A);
	delete[]A;
}
