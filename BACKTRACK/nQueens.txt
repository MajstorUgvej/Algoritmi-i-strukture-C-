#include <iostream>
#include <vector>

void Print(const std::vector<std::vector<int>>& M, int n)
{
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
			M[i][j] == 1 ? std::cout << "Q " : std::cout << ". ";
		std::cout << std::endl;
	}
	std::cout << std::endl;
}

bool Valid(const std::vector<std::vector<int>>& M, int n, int row, int col)
{
	for (int i = row - 1; i >= 0; i--) // For column
		if (M[i][col] == 1)
			return false;

	for (int i = 1; row - i >= 0 && col - i >= 0; i++) // For left diagonal
		if (M[row - i][col - i] == 1)
			return false;

	for (int i = 1; row - i >= 0 && col + i < n; i++) // For right diagonal
		if (M[row - i][col + i] == 1)
			return false;
	return true;
}

void N_Queens(std::vector<std::vector<int>>& M, int n, int row)
{
	if (row == n)
	{
		Print(M, n);
	}
	else
	{
		for (int i = 0; i < n; i++)
		{
			if (Valid(M, n, row, i))
			{
				M[row][i] = 1;
				N_Queens(M, n, row + 1);
				M[row][i] = 0;
			}
		}
	}
}

int main()
{
	int n;
	std::cout << "Enter n: ";
	std::cin >> n;

	std::vector<std::vector<int>> M(n, (std::vector<int>(n, 0)));

	N_Queens(M, n, 0);
}
