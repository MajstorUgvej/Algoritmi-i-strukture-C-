#include <iostream>
#include <vector>

/*
	comparisons: n
	swaps: n?
	adaptive: no
	stable: no
	extra memory: Auxilary array(max(n))
*/

//time:O(n) - mal moja muda al aj, space: O(n) -  to je isto laz
void CountSort(int niz[], int n)	// Koristan samo sa nizove u malom range-u, i da su brojevi pozitivni(ako se ne modifikuje)
{
	int max = -1, i;
	for (i = 0; i < n; i++)
		max = std::max(max, niz[i]);

	max += 1;
	std::vector<int> CSniz(max, 0);
	for (i = 0; i < n; i++)
		CSniz[niz[i]]++;

	i = 0;
	for (int j = 0; j < max; j++)
	{
		while (CSniz[j] > 0)
		{
			niz[i++] = j;
			CSniz[j]--;
		}
	}
}


void Display(int niz[], int n)
{
	for (int i = 0; i < n; i++)
		std::cout << niz[i] << " ";
	std::cout << std::endl;
}

int main()
{
	int niz[] = { 5, 8, 4, 2, 3, 7, 6, 1, 0, 9, 2 };
	int n = std::size(niz);
	CountSort(niz, n);
	Display(niz, n);

}
