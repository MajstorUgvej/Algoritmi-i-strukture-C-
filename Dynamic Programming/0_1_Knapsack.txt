#include <iostream>
#include <vector>

int p[] = { 0,8,10,3,9 };
int w[] = { 0,1,2,1,3 };

int R_Knap(int n, int m)
{
	if (n == 0 || m == 0)
		return 0;

	if (w[n] <= m)
	{
		int left = R_Knap(n - 1, m);
		int right = R_Knap(n - 1, m - w[n]) + p[n];
		return std::max(left, right);
	}
	else
	{
		return R_Knap(n - 1, m);
	}
}

int Mem_Knap(std::vector<std::vector<int>> &M, int n, int m)
{
	if (n == 0 || m == 0)
	{
		M[n][m] = 0;
		return 0;
	}

	if (M[n][m] != -1)
		return M[n][m];

	if (w[n] <= m)
	{
		int left = Mem_Knap(M, n - 1, m);
		int right = Mem_Knap(M, n - 1, m - w[n]) + p[n];
		M[n][m] = std::max(left, right);
	}
	else
	{
		M[n][m] = Mem_Knap(M, n - 1, m);
	}
	return M[n][m];
}

int DP_Knap(int n, int m)
{
	std::vector<std::vector<int>> M(n + 1, std::vector<int>(m + 1, 0));

	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			if (w[i] <= j)
				M[i][j] = std::max(M[i - 1][j], p[i] + M[i - 1][j - w[i]]);
			else
				M[i][j] = M[i - 1][j];
		}
	}

	std::cout << "Tabulation table:\n";
	for (int i = 0; i <= n; i++)
	{
		for (int j = 0; j <= m; j++)
		{
			std::cout << M[i][j] << " ";
			if (M[i][j] < 10)
				std::cout << " ";
		}
		std::cout << std::endl;
	}
	return M[n][m];
}

int main()
{
	int n = 4, m = 3;
	std::cout << "Max profit(recursion) is: " << R_Knap(n, m) << std::endl << std::endl;

	std::vector<std::vector<int>> M(n + 1, std::vector<int>(m + 1, -1));
	std::cout << "Max profit(memoaization) is: " << Mem_Knap(M, n, m) << std::endl;
	std::cout << "Memoaization table:\n";
	for (int i = 0; i <= n; i++)
	{
		for (int j = 0; j <= m; j++)
		{
			std::cout << M[i][j] << " ";
			if (M[i][j] < 10)
				std::cout << " ";
		}
		std::cout << std::endl;
	}
	std::cout << std::endl;
	std::cout << "Max profit(DP) is: " << DP_Knap(n, m) << std::endl;
}
